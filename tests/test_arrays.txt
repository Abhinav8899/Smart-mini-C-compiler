// Arrays: fill and sum
int arr[5];
int i;
int sum;

// Fill array with squares
for (i = 0; i < 5; i = i + 1) {
    arr[i] = i * i;
}

// Sum the array
sum = 0;
for (i = 0; i < 5; i = i + 1) {
    sum = sum + arr[i];
}

print(sum);
